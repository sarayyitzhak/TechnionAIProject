Explanations about files in this project:

# .\Assets\*.png - images 
from: "https://www.flaticon.com/free-stickers/chef"

# .\Assets\loading.gif - animation that we made :)

# .\DataFiles\CbsFiles\socio_economic_rank.xls - data file about socio-economic rank 
from: "https://www.cbs.gov.il/he/mediarelease/doclib/2019/246/24_19_246t1.xls"

# .\DataFiles\CbsFiles\religious_population.xlsx - data file about religious population
from: "https://www.cbs.gov.il/he/publications/doclib/2017/population_madaf/population_madaf_2021_12.xlsx"

# .\DataFiles\CbsFiles\streets_by_area_key.xlsx - data file contains street names by area
from: "https://www.cbs.gov.il/he/mediarelease/doclib/2022/026/%D7%A8%D7%97%D7%95%D7%91%D7%95%D7%AA%20%D7%A2%D7%99%D7%A7%D7%A8%D7%99%D7%99%D7%9D%20%D7%95%D7%A9%D7%9B%D7%95%D7%A0%D7%95%D7%AA%20%D7%9C%D7%90%D7%A1%202022.xlsx"

# .\DataFiles\CbsFiles\transition_area_key.xlsx - data file that converts old area code to new area code
from: "https://www.cbs.gov.il/he/mediarelease/doclib/2022/026/%D7%9E%D7%A4%D7%AA%D7%97%20%D7%9E%D7%A2%D7%91%D7%A8%20%D7%90%D7%A1%202011%20%D7%9C%D7%90%D7%A1%202022.xlsx"

# .\DataFiles\GoogleFiles\full-google-data.json - data file that we made by google api that contains all the google data (there is option to use this file or use the google api again).

# .\DataFiles\RestFiles\chromedriver.exe - driver that we use for selenium module
download chrome driver from: "https://chromedriver.chromium.org/downloads" to DataFiles\RestFiles folder, according to your chrome version.
current version: 120.0.6099.109
